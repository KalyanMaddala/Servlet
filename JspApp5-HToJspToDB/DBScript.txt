CREATE TABLE "SYSTEM"."STUDENT" 
   (	"SNO" NUMBER(4,0) NOT NULL ENABLE, 
	"SNAME" VARCHAR2(20 BYTE), 
	"ADRRS" VARCHAR2(20 BYTE), 
	"AVG" FLOAT(126), 
	 CONSTRAINT "STUDENT_PK" PRIMARY KEY ("SNO"));
	 
	 CREATE SEQUENCE  "SYSTEM"."SNO_SEQ"  MINVALUE 1 MAXVALUE 10000 INCREMENT BY 1 START WITH 41 CACHE 20 NOORDER  NOCYCLE ;
